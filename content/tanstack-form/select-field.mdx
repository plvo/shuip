## Built-in features

- **Type-safe options**: Options as `Record<string, string>` (key = label, value = field value)
- **Type-safe field names**: Autocomplete with `DeepKeys<TFormData>`
- **Validation**: Sync and async validation support
- **Full type inference**: Value types inferred from form schema

## Common use cases

<CodeHighlight code={`const form = useForm({
  defaultValues: {
    country: '',
    plan: '',
    region: '',
  },
  onSubmit: async ({ value }) => {
    await saveData(value)
  },
})

// Basic select
<SelectField
  form={form}
  name='country'
  options={{
    'United States': 'us',
    'United Kingdom': 'uk',
    'France': 'fr',
    'Germany': 'de',
  }}
  label='Country'
  placeholder='Select a country'
  description='Choose your country'
/>

// Required selection
<SelectField
  form={form}
  name='plan'
  options={{
    'Free': 'free',
    'Pro': 'pro',
    'Enterprise': 'enterprise',
  }}
  label='Subscription Plan'
  placeholder='Choose a plan'
  formProps={{
    validators: {
      onChange: ({ value }) =>
        !value ? 'Please select a plan' : undefined
    }
  }}
/>

// With async validation
<SelectField
  form={form}
  name='region'
  options={{
    'North America': 'na',
    'Europe': 'eu',
    'Asia Pacific': 'apac',
  }}
  label='Region'
  formProps={{
    validators: {
      onChangeAsync: async ({ value }) => {
        if (!value) return undefined
        const available = await checkRegionAvailability(value)
        return available ? undefined : 'This region is currently unavailable'
      }
    }
  }}
/>`} language='tsx' />
